JavaScript Array Methods

A comprehensive guide to JavaScript Array methods

Array.prototype.at()

at(index)

Description

Returns the element at the specified index. Supports negative indexing.

Example


        const arr = [10, 20, 30];
        console.log(arr.at(1)); // 20
        console.log(arr.at(-1)); // 30

Array.prototype.concat()

concat(value1, value2, ...)

Description

Merges two or more arrays. Returns a new array.

Example


        const arr1 = [1, 2];
        const arr2 = [3, 4];
        const result = arr1.concat(arr2);
        console.log(result); // [1, 2, 3, 4]

Array.prototype.filter()

filter(callback(element, index, array))

Description

Creates a new array with all elements that pass the test implemented by the provided function.

Example


        const numbers = [1, 2, 3, 4];
        const even = numbers.filter(num => num % 2 === 0);
        console.log(even); // [2, 4]

Array.prototype.copyWithin()

copyWithin(target, start, end)

Description

Copies part of the array to another location within the array.

Example


          const arr = [1, 2, 3, 4];
          arr.copyWithin(1, 2);
          console.log(arr); // [1, 3, 4, 4]

Array.prototype.entries()

entries()

Description

Returns an iterator object with key/value pairs.

Example


          const arr = ['a', 'b', 'c'];
          const iterator = arr.entries();
          console.log(iterator.next().value); // [0, 'a']

Array.prototype.every()

every(callback(element, index, array))

Description

Tests whether all elements pass the test provided by a function.

Example


          const numbers = [1, 2, 3, 4];
          const allPositive = numbers.every(num => num > 0);
          console.log(allPositive); // true

Array.prototype.fill()

fill(value, start, end)

Description

Fills all elements of the array with a static value.

Example


          const arr = [1, 2, 3, 4];
          arr.fill(0, 1, 3);
          console.log(arr); // [1, 0, 0, 4]

Array.prototype.find()

find(callback(element, index, array))

Description

Returns the first element that satisfies the provided testing function.

Example


                const numbers = [5, 12, 8, 130, 44];
                const found = numbers.find(num => num > 10);
                console.log(found); // 12

Array.prototype.filter()

filter(callback(element, index, array))

Description

Creates a new array with elements that pass the test implemented by the callback function.

                const numbers = [1, 2, 3, 4, 5];
                const even = numbers.filter(num => num % 2 === 0);
                console.log(even); // [2, 4]

Array.prototype.map()

map(callback(element, index, array))

Description

Creates a new array populated with the results of calling a provided function on every element in the array.

Example


                const numbers = [1, 2, 3, 4];
                const doubled = numbers.map(num => num * 2);
                console.log(doubled); // [2, 4, 6, 8]

Array.prototype.reduce()

reduce(callback(accumulator, element, index, array), initialValue)

Description

Executes a reducer function on each element of the array, resulting in a single output value.

Example


                const numbers = [1, 2, 3, 4];
                const sum = numbers.reduce((acc, num) => acc + num, 0);
                console.log(sum); // 10

Array.prototype.some()

some(callback(element, index, array))

Description

Tests whether at least one element in the array passes the test implemented by the callback function.

Example


                const numbers = [1, 2, 3, 4];
                const hasEven = numbers.some(num => num % 2 === 0);
                console.log(hasEven); // true

Array.prototype.slice()

slice(start, end)

Description

Returns a shallow copy of a portion of an array into a new array object.

Example


                const numbers = [1, 2, 3, 4];
                const sliced = numbers.slice(1, 3);
                console.log(sliced); // [2, 3]

Array.prototype.splice()

splice(start, deleteCount, item1, item2, ...)

Description

Changes the contents of an array by removing or replacing existing elements and/or adding new elements.

Example


                const numbers = [1, 2, 3, 4];
                numbers.splice(2, 1, 99);
                console.log(numbers); // [1, 2, 99, 4]
